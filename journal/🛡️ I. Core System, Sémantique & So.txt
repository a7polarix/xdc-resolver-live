JavaScript
import crypto from 'crypto';

export default async function handler(req, res) {
    if (req.method !== 'POST') {
        return res.status(405).json({ error: 'Méthode non autorisée. Utilisez POST.' });
    }

    const { player, difficulty = 10, targetQuest = 'global' } = req.body;

    if (!player) {
        return res.status(400).json({ error: 'Paramètre [player] manquant.' });
    }

    // 1. Génération d'un nombre hautement aléatoire entre 1 et 20 via CSPRNG
    const buffer = crypto.randomBytes(4);
    const uint32 = buffer.readUInt32BE(0);
    const d20Roll = (uint32 % 20) + 1;

    // 2. Calcul du résultat sémantique
    const isCriticalSuccess = d20Roll === 20;
    const isCriticalFailure = d20Roll === 1;
    const isSuccess = isCriticalSuccess ? true : (isCriticalFailure ? false : d20Roll >= difficulty);

    // 3. Génération de l'empreinte unique du lancer (pour journal.depin)
    const rollId = crypto.createHash('sha256')
        .update(`${player}-${targetQuest}-${d20Roll}-${Date.now()}`)
        .digest('hex');

    // 4. Payload sémantique retourné à l'interface Fleurs de Lys
    const responsePayload = {
        object: "depin.diceroll.event",
        id: `roll_${rollId.substring(0, 16)}`,
        timestamp: new Date().toISOString(),
        metadata: {
            player: player,
            quest: targetQuest,
            difficulty_class: difficulty
        },
        dice: {
            type: "d20",
            raw_value: d20Roll,
            modifier: 0 // Évolutif selon les attributs du joueur plus tard
        },
        resolution: {
            success: isSuccess,
            critical_success: isCriticalSuccess,
            critical_failure: isCriticalFailure,
            summary: isCriticalSuccess ? "CRITICAL_SUCCESS" : (isCriticalFailure ? "CRITICAL_FAILURE" : (isSuccess ? "SUCCESS" : "FAILURE"))
        }
    };

    return res.status(200).json(responsePayload);
}

    // 1. Vérification si le domaine est une racine connue
    if (MESH_CONFIG_ROOTS[domainName]) {
        return renderStaticNode(domainName, MESH_CONFIG_ROOTS[domainName]);
    }
    
    // 2. Si non listé (Un des 6000 sous-domaines), génération procédurale via le Ledger
    return await generateProceduralGhostNode(domainName);
}

async function generateProceduralGhostNode(subDomain) {
    // On génère un hash unique à partir du nom du sous-domaine
    const hash = btoa(subDomain).substring(0, 8); 
    const charCodeSum = [...hash].reduce((acc, char) => acc + char.charCodeAt(0), 0);
    
    // Détermination de la faction du nœud fantôme (Bleu ou Rouge)
    const alignment = charCodeSum % 2 === 0 ? "ROUGE_CHAOS" : "BLEU_ORDRE";
    const dangerLevel = (charCodeSum % 10) + 1; // Danger de 1 à 10
    
    // Calcul de l'impact de la Peur (Loi d'entropie)
    const hpDrain = dangerLevel * 1.5;

    // Structure renvoyée dynamiquement à l'interface vectorielle
    return {
        node_type: "GHOST_LABYRINTH",
        lore: `Vous avez glissé dans les failles d'Arcadia. Ce lieu porte la signature spectrale : [${hash}].`,
        mechanics: `Infection par la Peur élevée. Perte de ${hpDrain} HP par bloc. Résolvez l'énigme pour fuir.`,
        next_nodes: ['arcadia.depin', 'deathnote.depin'], // Portes de sortie standards
        vector_layout: {
            color: alignment === "ROUGE_CHAOS" ? "#ff0055" : "#0055ff",
            layer: "sub_catacombs",
            effect: "noise_vibration_distortion",
            mesh_density: charCodeSum % 100
        },
        enigma: {
            question: `Convertissez le fragment simeon [${hash}] sous la grammaire unicode.depin pour annuler la Peur.`,
            difficulty: charCodeSum % 20
        }
    };
}

        // MAQUETTAGE DE LA GRILLE VECTORIELLE (mathematics.depin)
        function buildGrid() {
            const grid = document.getElementById("math-grid");
            grid.innerHTML = "";
            const width = 800, height = 400;
            for(let i=0; i<width; i+=40) {
                let line = document.createElementNS("http://www.w3.org/2000/svg", "line");
                line.setAttribute("x1", i); line.setAttribute("y1", 0);
                line.setAttribute("x2", i); line.setAttribute("y2", height);
                grid.appendChild(line);
            }
            for(let j=0; j<height; j+=40) {
                let line = document.createElementNS("http://www.w3.org/2000/svg", "line");
                line.setAttribute("x1", 0); line.setAttribute("y1", j);
                line.setAttribute("x2", width); line.setAttribute("y2", j);
                grid.appendChild(line);
            }
        }

        // ANIMATION OSCILLANTE (spectrum.depin)
        function animateSpectrum() {
            const wave = document.getElementById("spectrum-wave");
            let count = 0;
            setInterval(() => {
                count += 0.1;
                let d = "M 0 40 ";
                for (let x = 0; x < 350; x += 30) {
                    let y = 40 + Math.sin(count + (x * 0.05)) * 20;
                    d += `Q ${x + 15} ${y + (playerState.isBerserk ? 15 : 0)}, ${x + 30} 40 `;
                }
                wave.setAttribute("d", d);
                if(playerState.isBerserk) {
                    wave.setAttribute("stroke", "#ff0033");
                } else {
                    wave.setAttribute("stroke", "#00ffcc");
                }
            }, 50);
        }

        // GESTION DU LIVRE ET DES CHOIX NARRATIFS
        function renderScene(key) {
            const scene = storyTree[key];
            if (!scene) return;

            playerState.activeQuest = key;
            document.getElementById("quest-node").innerText = key.toUpperCase();

            // Exécution des hooks spécifiques à l'entrée de scène
            if (scene.onEnter) scene.onEnter();

            // Affichage du texte
            document.getElementById("narrative-text").innerHTML = scene.text;

            // Construction des boutons de choix
            const container = document.getElementById("choices-container");
            container.innerHTML = "";

            scene.choices.forEach(choice => {
                const btn = document.createElement("button");
                btn.className = "choice-btn";
                btn.innerText = choice.text;
                
                btn.onclick = () => {
                    if (choice.target) {
                        renderScene(choice.target);
                    } else if (choice.action) {
                        window[choice.action]();
                    }
                };
                container.appendChild(btn);
            });
        }

        function setupChallenge(node, difficulty, successTarget, failTarget) {

            playerState.pendingResolution = {

        function updateNodeDisplay(node) {

            // Effet graphique : déplace visuellement le marqueur central aléatoirement pour feindre le saut de réseau
            const cx = 30 + Math.random() * 40;
            const cy = 30 + Math.random() * 40;
            document.getElementById("node-center").setAttribute("cx", cx + "%");
            document.getElementById("node-center").setAttribute("cy", cy + "%");
            document.getElementById("current-node-tag").setAttribute("x", cx + "%");
            document.getElementById("current-node-tag").setAttribute("y", (cy - 4) + "%");
        }

        // MOTEUR DU HASARD SÉMANTIQUE : diceroll.depin
        function executeRoll() {
            if (!playerState.pendingResolution) return;

            const diceOverlay = document.getElementById("d20-overlay");
            const diceText = document.getElementById("d20-text");
            diceOverlay.className = "rolling";
            logJournal("Appel d'entropie envoyé à diceroll.depin...", "system");

            setTimeout(() => {
                diceOverlay.className = "";
                
                // Formule de Cohérence : hasard cryptographique CSPRNG simulé + modificateurs de masse Janus
                let rawRoll = Math.floor(Math.random() * 20) + 1;
                let modifier = 0;

                // Application des lois physiques de physics.depin
                if (playerState.currentNode === "tartaria.depin") {
                    modifier = -3; // Masse négative / Peur provoque un malus gravitationnel sur le dé
                    logJournal("physics.depin applique une distorsion Janus de -3 (Masse Négative).", "system");
                }

                let finalRoll = Math.max(1, Math.min(20, rawRoll + modifier));
                diceText.textContent = finalRoll;

                const challenge = playerState.pendingResolution;
                playerState.pendingResolution = null;

                logJournal(`Diceroll Résolu. Jet Brut: ${rawRoll} | Modificateur: ${modifier} | Résultat: ${finalRoll} (Seuil requis: ${challenge.difficulty})`, "system");

                // Arbitrage des succès / échecs critiques et standards
                if (rawRoll === 1) {
                    logJournal("ÉCHEC CRITIQUE ON-CHAIN ! Rupture du bouclier sémantique.", "fail");
                    renderScene(challenge.fail);
                } else if (rawRoll === 20 || finalRoll >= challenge.difficulty) {
                    logJournal(`Succès validé contre le seuil ${challenge.difficulty}!`, "success");
                    renderScene(challenge.success);
                } else {
                    logJournal(`Échec du test de compétence. Les validateurs rejettent la preuve.`, "fail");
                    renderScene(challenge.fail);
                }

            }, 800);
        }

        function damagePlayer(amount) {
            playerState.hp = Math.max(0, playerState.hp - amount);
            document.getElementById("hp-display").innerText = `${playerState.hp}/100`;
            document.getElementById("hp-bar").style.width = `${playerState.hp}%`;
            logJournal(`Drain de fluide vital : -${amount} HP subis.`, "fail");
            if (playerState.hp <= 0 && playerState.activeQuest !== "game_over") {
                renderScene("game_over");
            }
        }

        function adjustEnergy(amount) {
            playerState.energy = Math.round((playerState.energy + amount) * 1000) / 1000;
            document.getElementById("energy-display").innerText = `${playerState.energy} J`;
            logJournal(`Conversion thermodynamique effectuée. Énergie réseau: ${playerState.energy} Joules.`, "system");
        }
